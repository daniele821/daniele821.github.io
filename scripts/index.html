Run the following to do install operations:

- FEDORA-KDE:

        sudo dnf --assumeyes install git &&
        TMP_DIR="$(mktemp -d)" &&
        REPO_DIR="${TMP_DIR}/dotfiles" &&
        git clone https://github.com/daniele821/dotfiles "${REPO_DIR}" &&
        git -C "$REPO_DIR" switch fedora-kde &&
        BRANCH= "${TMP_DIR}/dotfiles/autosaver" run ||
        echo -e "\e[1;31mfailed to download and run init scripts\e[m"
